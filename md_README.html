Tutorial

1 $ export GITHUB_USERNAME=<имя_пользователя>
2 $ export GITHUB_TOKEN=<полученный_токен>
1 $ git clone https://github.com/${GITHUB_USERNAME}/lab04 lab07
2 $ cd lab07
3 $ git remote remove origin
4 $ git remote add origin https://github.com/${GITHUB_USERNAME}/lab07
1 $ cat > .travis.yml <<EOF
2 language: cpp
3 EOF
1 $ cat >> .travis.yml <<EOF
2 
3 script:
4 - cmake -H. -B_build -DCMAKE_INSTALL_PREFIX=_install
5 - cmake --build _build
6 - cmake --build _build --target install
7 EOF
1 $ cat >> .travis.yml <<EOF
2 
3 addons:
4  apt:
5  sources:
6  - george-edison55-precise-backports
7  packages:
8  - cmake
9  - cmake-data
10 EOF
1 $ travis login --github-token ${GITHUB_TOKEN}
1 $ travis lint
1 $ ex -sc '1i|<фрагмент_вставки_значка>' -cx README.md
1 $ git add .travis.yml
2 $ git add README.md
3 $ git commit -m"added CI"
4 $ git push origin master
1 $ travis lint
2 $ travis accounts
3 $ travis sync
4 $ travis repos
5 $ travis enable
6 $ travis whatsup
7 $ travis branches
8 $ travis history
9 $ travis show
